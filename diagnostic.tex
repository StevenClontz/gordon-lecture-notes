\question[5]
Find $\frac{dz}{dt}$ given $z=e^{2t}$.

\begin{checkboxes}
\choice $2\ln t$
\choice $\frac{1}{2}\ln t$
\CorrectChoice $2e^{2t}$
\choice $\frac{1}{2}e^{2t}$
\choice None of these
\end{checkboxes}

\vfill

\question[5]
Simplify $\sqrt{16\sin^2 t+9\sin^2 t + 25\cos^2 t}$.

\begin{checkboxes}
\CorrectChoice $5$
\choice $5\tan t$
\choice $\sqrt{50}$
\choice $\cos{5t}$
\choice None of these
\end{checkboxes}

\vfill

\choice $-2$
\choice $4$
\choice None of these
\end{checkboxes}

\question[5]

\choice $x^2e^x+C$

\choice None of these
\end{checkboxes}

\question[5]

\CorrectChoice $e^x\sec^2 x + e^x\tan x$
\choice None of these
\end{checkboxes}
